def model_builder(inp_shape, out_shape):
    model = Sequential()
    model.add(Dense(256, input_dim=inp_shape, activation=relu))
    model.add(Dense(128, activation=relu))
    model.add(Dense(out_shape, activation=linear))
    model.compile(loss='mse', optimizer='Adam')
    return model

def get_action(eps, state):
    if np.random.rand() <= eps:
        action = int(random.randrange(n_actions))
        return action
    else:
        logits = model(state)
        action = int(np.argmax(logits[0]))
    return action

def weights_update(memory, action, state, new_state, reward, done, info):
    if len(memory["state"]) > 0:
        memory["state"] = np.vstack((memory["state"], state))
        memory["new_state"] = np.vstack((memory["new_state"], new_state))
    else:
        memory["state"] = np.array(state)
        memory["new_state"] = np.array(new_state)
    memory["action"] = np.append(memory["action"], action)
    memory["reward"] = np.append(memory["reward"], reward)
    memory["done"] = np.append(memory["done"], done)

    state = new_state

    if len(memory["state"]) >= batch_size:
        idxs = np.random.randint(len(memory["done"]), size=batch_size)
        batch = {
                    "state": np.squeeze(memory["state"][idxs]), 
                    "action": memory["action"][idxs], 
                    "reward": memory["reward"][idxs], 
                    "new_state": np.squeeze(memory["new_state"][idxs]), 
                    "done": memory["done"][idxs]
                }
                
        q_values = batch["reward"] + discount_rate*(np.amax(model.predict_on_batch(batch["new_state"]), 
                                                                    axis=1))*(1-batch["done"])
        network_training = model.predict_on_batch(batch["state"])
        network_training[np.array(range(batch_size)), batch["action"]] = q_values
        model.fit(batch["state"], network_training, epochs = 1, verbose = 0)
    return memory, state, model

def train_model(env, n_states, n_actions, iters, model, batch_size, discount_rate, eps = 1):
    scores = np.array([])
    memory = {"state":np.array([]), 
            "action":np.array([], dtype = int), 
            "reward":np.array([]),
            "new_state":np.array([]), 
            "done":np.array([])
            }
    check = 100
    
    for episode in range(iters):
        
        if episode == check:
            last_negative = np.argwhere(scores<0)[-1]
            positive_tail = episode - last_negative
            if positive_tail >= 100:
                break
            else:
                check = last_negative + 100
                
        state = env.reset()
        state = np.reshape(state, (1, n_states))
        game_score = 0
        max_steps = 1000 #?
        
        for i in range(max_steps):
            action = get_action(eps, state) 
            new_state, reward, done, info = env.step(action)
            new_state = new_state.reshape([1, n_states])  
            game_score += reward
            memory, state, model = weights_update(memory, action, state, new_state, reward, done, info)
                
            if eps > 0.01:
                eps -= 0.01
            
            if done:
                break
                
        scores = np.append(scores, game_score)
        print('Score for episode {}: {}'.format(episode, scores[-1]))
        
    return scores, model
